cd sniffer
gcc main.c -o sniffer
sudo ./sniffer -i eth0     # Sniff on eth0
sudo ./sniffer -i          # Interactive interface selection
sudo ./sniffer             # Auto-select default interface
sudo ./sniffer -h          # Help menu
